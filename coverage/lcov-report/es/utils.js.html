function extractPageInfo(url, appUtm = {}) {
    let match = url.match(/([^?]+)(\?.+)?/);
    let [, pagePath = '', queryString = ''] = match || [];
    if (pagePath && pagePath[0] === '/') {
        pagePath = pagePath.slice(1);
    }
    queryString = new URLSearchParams(queryString);
    const pageId = queryString.get('__pageId');
    queryString.delete('__pageId');
    const isWorker = queryString.get('__worker') === 'true';
    queryString.delete('__worker');
    const isNative = queryString.get('__native') === 'true';
    queryString.delete('__native');
    let queryStringResult = queryString.toString();
    // URLSearchParams automatically add = into end of string
    // so we remove it, and decodeURIComponent
    // this is fix bug for case of Infina
    // pages/index?a
    // pages/index?https://tiki.vn
    if (queryStringResult.endsWith('=')) {
        queryStringResult = decodeURIComponent(queryStringResult.slice(0, queryStringResult.length - 1));
    }
    else {
        // update queryString with data from appUtm
        Object.keys(appUtm).forEach((key) => {
            if (!queryString.has(key)) {
                queryStringResult = `${queryStringResult}${queryStringResult ? '&' : ''}${key}=${appUtm[key]}`;
            }
        });
    }
    return {
        pageId: pageId || '0',
        pagePath,
        queryString: queryStringResult,
        isWorker,
        isNative,
    };
}
exports.extractPageInfo = extractPageInfo;
function getPageInfo() {
    const { href } = location;
    let match = href.match(/#(.*)/);
    const [, url] = match;
    return extractPageInfo(url);
}

function parseQueryToObject(queryString) {
    let referrerInfo = {};
    let parsedData = {};
    try {
        parsedData = qs.parse(decodeURIComponent(queryString));
        if (parsedData.referrerInfo) {
            referrerInfo = JSON.parse(parsedData.referrerInfo);

    return {
        query: Object.assign(Object.assign({}, parsedData), referrerInfo.extraData),

    };
}

function getSemVer(value) {
    const [major, minor, patch] = (value || '').split('.');
    return {
        major: parseInt(major, 10) || 0,
        minor: parseInt(minor, 10) || 0,
        patch: parseInt(patch, 10) || 0,
    };
}

function isSemVerAfter(as, bs) {
    if (as.major > bs.major) {
        return true;
    }
    if (as.major < bs.major) {
        return false;
    }
    if (as.minor > bs.minor) {
        return true;
    }
    if (as.minor < bs.minor) {
        return false;
    }
    return as.patch >= bs.patch;
}
