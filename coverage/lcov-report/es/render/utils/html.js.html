const convertStyleTextObject = (style) => {
    if (!style.length) {
        return null;
    }
    const parts = style.split(/;\s*/g).sort();
    const object = {};
    for (let i = 0; i < parts.length; i++) {
        const kv = parts[i];
        if (kv) {
            const key = kv.split(/:\s*/)[0];
            let value = kv.slice(key.length + 1).trim();
            if (key === 'background-position') {
                switch (value) {
                    case 'center':
                    case 'top':
                    case 'bottom':
                        value = `center ${value}`;
                        break;
                    case 'left':
                    case 'right':
                        value = `${value} center`;
                        break;
                }
            }
            object[key] = value;
        }
    }
    return object;
};
