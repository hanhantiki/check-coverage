import { initialize as initializeReactDevToolsBackend } from 'react-devtools-inline/backend';
import Bridge from '@tikivn/miniapp-shared/lib/Bridge';
import DevToolsBackendConnection from './DevToolsBackendConnection';
 
if (window.__DEVTOOLS_URL__) {
  // initialize __REACT_DEVTOOLS_GLOBAL_HOOK__
  // NOTE: MAKE SURE THIS IS RUN BEFORE REACT DOM IS RUN
  initializeReactDevToolsBackend(window);
  // setup connection with React Backend
  const devToolsBackendConn = new DevToolsBackendConnection(window);
 
  let bridgeResolve;
  const bridgeReady = new Promise((resolve) => (bridgeResolve = resolve));
 
  let documentUpdatedTimer;
  let isActive = true;
 
  const hook = window.__REACT_DEVTOOLS_GLOBAL_HOOK__;
  hook.sub('resume', () => {
    isActive = true;
    bridgeReady.then((bridge) => {
      documentUpdatedTimer = setTimeout(() => {
        bridge.send('event', {
          method: 'DOM.documentUpdated',
        });
      }, 500);
    });
  });
 
  hook.sub('pause', () => {
    clearTimeout(documentUpdatedTimer);
    isActive = false;
  });
 
  const { href } = window.location;
  const match = href.match(/__pageId=(\d+)(?:&|$)/);
  const pageId = (match && match[1]) || '';
  const ws = new WebSocket(`${window.__DEVTOOLS_URL__}?role=render&pageId=${pageId}`);
  ws.onopen = () => {
    const bridge = new Bridge({
      listen(fn) {
        ws.onmessage = ({ data }) => {
          if (isActive) {
            const object = JSON.parse(data);
            fn({ event: 'request', params: object });
          }
        };
        return () => {
          ws.onmessage = null;
        };
      },
      send(_, data) {
        if (isActive) {
          ws.send(JSON.stringify(data));
        }
      },
    });
    bridgeResolve(bridge);
    bridge.addListener('request', (request) => {
      const result = devToolsBackendConn.devToolsRequest(request);
      const { id } = request;
      bridge.send('result', { id, result });
    });
 
    devToolsBackendConn.onDevToolsDomainEvent((event) => {
      const { data } = event;
      bridge.send('event', data);
    });
  };
}
