// Patching for Chrome DevTools Frontend to hide features and add miniap extensions.
const getPanelRef = (name) => {
  const nodeList = document.querySelectorAll('iframe');
  let target = null;
  for (let i = 0, num = nodeList.length; i < num; i++) {
    if (nodeList[i].src.indexOf(`/${name.toLowerCase()}/${name.toLowerCase()}.html`) !== -1) {
      target = nodeList[i];
      break;
    }
  }
  return target;
};
 
const domainListeners = {};
const sendPanelMessage = (panelName, message) => {
  const panel = getPanelRef(panelName);
  const messageStr = typeof message === 'string' ? message : JSON.stringify(message);
  if (panel) {
    panel.contentWindow.postMessage(messageStr, '*');
  }
};
 
const createPanel = (info, port) => {
  const { title, page, domain } = info;
  const id = `extension-panel-${page}`;
  const message = { id, title, page };
  Extensions.extensionServer._onCreatePanel(message, port);
  if (!domainListeners[domain]) {
    domainListeners[domain] = (_message) => sendPanelMessage(domain, _message);
  }
};

const createMiniAppMessageObserver = () => {
  const mainConn = SDK.targetManager.mainTarget().router().connection();
  const { _onMessage } = mainConn;
  mainConn.setOnMessage((message) => {
    try {
      const data = JSON.parse(message);
      const { method, params } = data;
      const splitPoint = method.indexOf('.');
      const domain = method.slice(0, splitPoint);
      if (domainListeners[domain]) {
        domainListeners[domain](params);
        return;
      }
    } catch (e) {}
    _onMessage(message);
  });
};
 
const mainTargetListener = () => {
  if (SDK.targetManager.mainTarget()) {
    createMiniAppMessageObserver();
    MiniAppSimulator.devToolsFrontendReady();
    return;
  }
 
  const targetObserver = {
    targetAdded: () => {
      SDK.targetManager.unobserveTargets(targetObserver);
      createMiniAppMessageObserver();
      MiniAppSimulator.devToolsFrontendReady();
    },
    targetRemoved: () => {
      console.warn('[devtools] targetObserver targetRemoved');
    },
  };
  SDK.targetManager.observeTargets(targetObserver);
};
 
// hook to capture message from DevTools Frontend to Chrome DevTools Backend
window.messageWriter = async (message, conn) => {
  const data = JSON.parse(message);
  const { method } = data;
  if (method) {
    // we will handle below domains instead of let DevTools Backend handle it.
    const [domain] = method.split('.');
    if (['Page.enable', 'DOM.enable', 'CSS.enable', 'Overlay.enable'].includes(method)) {
      const msg = JSON.stringify({ id: data.id, result: {} });
      return conn._onMessage.call(null, msg);
    } else if (method === 'Network.getResponseBody') {
      const object = JSON.parse(message);
      const {
        params: { requestId },
      } = object;
      const result = (await MiniAppSimulator.getNetworkResponseBody(requestId)) || {};
      conn._onMessage.call(null, JSON.stringify({ id: data.id, result }));
      return;
    } else if (['DOM', 'Overlay', 'CSS'].includes(domain)) {
      if (data.method === 'DOM.getDocument') {
        data.params.depth = -1;
        data.params.pierce = true;
        message = JSON.stringify(data);
      }
      MiniAppSimulator.devToolsRequest(message);
      return;
    }
  }
  conn._socket.send(message);
};

window.messageReader = (message, conn) => {
  // avoid devtools frontend flush console logs
  const object = JSON.parse(message);
  object.error = null;
  conn._onMessage.call(null, object);
};

// Wait for PORT connect between simulator and DevTools.
// when devtools is ready, IDE will send a port to open connection between devtools and simulator/debuggerWorker.
window.addEventListener('message', (event) => {
  const { origin, data, ports } = event;
  // IDE open window with file://, we use this to indicate message from IDE.
  if (origin.startsWith('file://')) {
    if (data.kind === 'initIDEPort') {
      const [port] = ports;
      MiniAppSimulator.setIDEPort(port);
    }
  }
});
