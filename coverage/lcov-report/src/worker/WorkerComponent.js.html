import ComponentRegistry from '../ComponentRegistry';
import lodash from 'lodash';
import getLogger from '../logging';
 
const logger = getLogger('worker').getChildLogger('WorkerComponent');
 
export default class WorkerComponent {
  constructor(page, is, id, config) {
    this.page = page;
    this.id = id;
    this.is = is;
    const { user: userConfig, system: systemConfig } = ComponentRegistry.getConfig(is);
 
    // initialize publicInstance
    const publicInstance = Object.create(
      lodash.cloneDeep({
        ...userConfig,
        ...systemConfig,
      }),
    );
    publicInstance.is = is;
    publicInstance.$id = id;
    publicInstance.$page = page.publicInstance;
    publicInstance.setData = (...args) => {
      this.setData(...args);
    };
    if (typeof publicInstance.data === 'function') {
      publicInstance.data = publicInstance.data() || {};
    }
    publicInstance.data = publicInstance.data || {};
    publicInstance.props = { ...(userConfig.props || {}), ...(config || {}) };
    const { methods } = userConfig;
    methods &&
      Object.keys(methods).forEach((f) => {
        if (!publicInstance.hasOwnProperty(f)) {
          publicInstance[f] = methods[f];
        }
      });
 
    this.prevData = publicInstance.data;
    this.publicInstance = publicInstance;
    this.fireComponentLifeCycleEvent('onInit');
    this.setComponentConfig(config);
  }
 
  setData(data, callback) {
    if (this.unloaded) {
      logger.warn('setData(...) can only update a mounted component.');
    } else {
      const {
        publicInstance: { data: currentData },
      } = this;
      const _prevData = this.prevData;
      const _prevProps = this.publicInstance.props;
      this.prevData = currentData;
      this.publicInstance.data = Object.assign(currentData, data);
      this.page.setRemoteData(
        {
          data,
          componentType: 'component',
          componentId: this.id,
        },
        () => {
          if (callback) {
            callback();
          }
        },
      );
      this.fireComponentLifeCycleEvent('didUpdate', _prevProps, _prevData);
    }
  }
 
  setComponentConfig(config) {
    const _prevData = this.prevData;
    const _prevProps = this.publicInstance.props;
    function compare(props1, props2) {
      // trim function
      try {
        const _props1 = JSON.parse(JSON.stringify(props1));
        const _props2 = JSON.parse(JSON.stringify(props2));
        return lodash.isEqual(_props1, _props2);
      } catch (e) {
        return false;
      }
    }
    const nextProps = this.normalizeProps(config || {});
    this.publicInstance.props = nextProps;
    if (!compare(nextProps, _prevProps)) {
      this.fireComponentLifeCycleEvent('didUpdate', _prevProps, _prevData);
      this.fireComponentLifeCycleEvent('deriveDataFromProps', nextProps);
    }
  }
 
  normalizeProps(props) {
    const _props = {};
    Object.keys(props).forEach((att) => {
      if (att.match(/^(on|catch)[A-Z]/)) {
        _props[att] = this.page.getEventHandler(props[att]);
      } else {
        _props[att] = props[att];
      }
    });
    return { ...(this.publicInstance.props || {}), ..._props };
  }
 
  ready() {
    this.fireComponentLifeCycleEvent('didMount');
  }
 
  update() {
    this.fireComponentLifeCycleEvent('didUpdate');
  }
 
  unload() {
    this.unloaded = true;
    this.fireComponentLifeCycleEvent('didUnmount');
  }
 
  fireComponentLifeCycleEvent = (events, ...args) => {
    let eventArr = events;
    if (!Array.isArray(events)) {
      eventArr = [events];
    }
    eventArr.forEach((event) => {
      try {
        const { publicInstance } = this;
        if (event in publicInstance) {
          return publicInstance[event].apply(publicInstance, args);
        }
      } catch (e) {
        logger.error(
          `[worker][WorkerComponent] Failed to trigger event ${event} ${e.message}`,
          e.stack,
        );
      }
    });
  };
}
