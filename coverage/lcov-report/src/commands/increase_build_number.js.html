import path from 'path';
import fs from 'fs-extra';
import os from 'os';
import logUtil from '../logUtil';
 
export default function increaseBuildNumberCommand([folder = '.'], { build_number, version } = {}) {
  const cwd = path.resolve(folder);
  const pkgJsonPath = path.join(cwd, 'package.json');
  const pkg = JSON.parse(fs.readFileSync(pkgJsonPath).toString());
  pkg.version = version || pkg.version || '0.1.0';
  const { tiki = {}, name = '' } = pkg;
  tiki.buildNumber =
    build_number || build_number === 0 ? build_number : parseInt(tiki.buildNumber || 0, 10) + 1;
  const packageJson = {
    name,
    version,
    tiki,
    ...pkg,
  };
  fs.writeFileSync(path.join(cwd, 'package.json'), JSON.stringify(packageJson, null, 2) + os.EOL);
 
  logUtil.printAppDetails(cwd);
}
