import fs from 'fs';
import path from 'path';
import parseJson from 'json-parse-better-errors';
import { findProjectRoot, normalizePathForWin } from './utils/path';
 
function getAbsoluteComponentConfigPath(srcDir, pagePath, componentPath) {
  // ./path or ../path are relative
  // path or /path means root
  const isRelative = componentPath.startsWith('.');
  const nodeModulesDir = path.resolve(srcDir, '../node_modules');
  const root = path.resolve(srcDir, '../');
  const filePatterns = isRelative
    ? [
        path.join(srcDir, path.dirname(pagePath), `${componentPath}.json`),
        path.join(srcDir, path.dirname(pagePath), componentPath, 'index.json'),
        path.resolve(root, path.dirname(pagePath), `${componentPath}.json`),
        path.resolve(root, path.dirname(pagePath), componentPath, 'index.json'),
        path.resolve(nodeModulesDir, path.dirname(pagePath), `${componentPath}.json`),
        path.resolve(nodeModulesDir, path.dirname(pagePath), componentPath, 'index.json'),
      ]
    : [
        path.join(srcDir, `${componentPath}.json`),
        path.join(srcDir, componentPath, 'index.json'),
        path.resolve(root, `${componentPath}.json`),
        path.resolve(root, componentPath, 'index.json'),
        path.resolve(nodeModulesDir, `${componentPath}.json`),
        path.resolve(nodeModulesDir, componentPath, 'index.json'),
      ];
  const configPath = filePatterns.find((pattern) => fs.existsSync(path.resolve(pattern)));
  return configPath;
}
 
function getComponentConfig(srcDir, pagePath, componentPath) {
  const configPath = getAbsoluteComponentConfigPath(srcDir, pagePath, componentPath);
  if (configPath) {
    try {
      const componentJson = JSON.parse(fs.readFileSync(configPath));
      const { component } = componentJson;
      if (component === true) {
        let entryPath = configPath
          .slice(0, -path.extname(configPath).length)
          .replace(`${srcDir}${path.sep}`, '');
        if (configPath.includes('node_modules')) {
          const root = path.resolve(srcDir, '../');
          entryPath = configPath
            .slice(0, -path.extname(configPath).length)
            .replace(`${root}${path.sep}`, '');
        }
        return normalizePathForWin(entryPath);
      }
    } catch (e) {
      throw new Error(`JSONError: ${e.message} in ${componentPath}.json`);
    }
  }
  return null;
}
 
function getComponentConfigValue(srcDir, pagePath, componentPath) {
  const configPath = getAbsoluteComponentConfigPath(srcDir, pagePath, componentPath);
  if (configPath) {
    try {
      const componentJson = parseJson(fs.readFileSync(configPath).toString('utf-8'));
      return componentJson;
    } catch (e) {
      throw new Error(`JSONError: ${e.message} in ${componentPath}.json`);
    }
  }
  return {};
}
 
function getPageConfig(srcDir, pagePath) {
  const fullPageJsonPath = `${path.join(srcDir, pagePath)}.json`;
  if (fs.existsSync(fullPageJsonPath)) {
    try {
      // this might throw error, let it throw so compiler can catch it detail.
      const pageJson = parseJson(fs.readFileSync(fullPageJsonPath).toString('utf-8'));
      // process usingComponents field
      const { usingComponents } = pageJson;
      if (usingComponents && typeof usingComponents === 'object') {
        const normalizedUsingComponents = Object.keys(usingComponents).reduce(
          (ret, componentName) => {
            const componentPath = usingComponents[componentName];
            const componentEntry = getComponentConfig(srcDir, pagePath, componentPath);
            return {
              ...ret,
              ...(componentEntry && { [componentName]: componentEntry }),
            };
          },
          {},
        );
        if (Object.keys(normalizedUsingComponents).length > 0) {
          pageJson.usingComponents = normalizedUsingComponents;
        } else {
          delete pageJson.usingComponents;
        }
      }
      return pageJson;
    } catch (e) {
      throw new Error(`JSONError: ${e.message} in ${pagePath}.json`);
    }
  }
  // we allows missing page.json, so don't need throw error here
  return {};
}
 
function getAppJsonConfig(srcDir) {
  const appPath = path.join(srcDir, 'app.json');
  if (!fs.existsSync(appPath)) {
    throw new Error('Invalid miniapp project. We could not find app.json in your workspace.');
  }
  const appFileContent = fs.readFileSync(appPath).toString('utf-8');
  try {
    return parseJson(appFileContent);
  } catch (e) {
    throw new Error(`JSONError: ${e.message} in app.json`);
  }
}
 
function getAppConfig({ srcDir }) {
  const defaultAppJson = {
    window: {
      defaultTitle: 'My App',
    },
  };
  const appConfig = {};
  const appJson = getAppJsonConfig(srcDir);
 
  // copy tabBar icon to output folder
  const { pages: origPages = [] } = appJson;
  const pages = origPages.reduce((ret, pagePath) => {
    let normalizedPagepPath = pagePath;
    if (normalizedPagepPath.startsWith('./')) {
      normalizedPagepPath = normalizedPagepPath.slice(2);
    }
    if (normalizedPagepPath.startsWith('/')) {
      normalizedPagepPath = normalizedPagepPath.slice(1);
    }
 
    return [...ret, normalizedPagepPath];
  }, []);
  appConfig.app = {
    ...defaultAppJson,
    ...appJson,
    pages,
  };
  pages.forEach((pagePath) => {
    appConfig[pagePath] = getPageConfig(srcDir, pagePath);
  });
  return appConfig;
}
function resolveComponentConfig(pageConfig, usingComponents, components, srcDir, rootPath) {
  if (pageConfig.usingComponents) {
    Object.values(pageConfig.usingComponents).forEach((componentPath) => {
      const componentConfig = getComponentConfigValue(srcDir, rootPath, componentPath);
      if (componentConfig.usingComponents) {
        const abComponentPath = getComponentConfig(srcDir, rootPath, componentPath);
        usingComponents[abComponentPath] = Object.keys(componentConfig.usingComponents).reduce(
          (n, k) => {
            const p = getComponentConfig(
              srcDir,
              abComponentPath,
              componentConfig.usingComponents[k],
            );
            if (p) {
              n[k] = p;
              components[p] = getComponentConfigValue(srcDir, componentPath, p);
            }
            return n;
          },
          {},
        );
      }
      const p = getComponentConfig(srcDir, rootPath, componentPath);
      components[p] = componentConfig;
      resolveComponentConfig(componentConfig, usingComponents, components, srcDir, componentPath);
    });
  }
}
 
function copyIcon(icon, srcDir, output, outputFileSystem) {
  if (!fs.existsSync(path.join(srcDir, icon))) {
    return;
  }
  const outputPath = path.join(output, icon);
  const outputFolder = path.dirname(outputPath);
  if (!outputFileSystem.existsSync(outputFolder)) {
    outputFileSystem.mkdirSync(outputFolder, { recursive: true });
  }
  const iconPath = path.join(srcDir, icon);
  // hack to create file
  outputFileSystem.writeFileSync(outputPath, '');
  fs.createReadStream(iconPath).pipe(outputFileSystem.createWriteStream(outputPath));
}
 
export default function generateAppConfig(opts) {
  const { cwd = process.cwd(), baseDir } = opts;
 
  let srcDir = baseDir;
  if (!baseDir) {
    srcDir = findProjectRoot(cwd);
  }
 
  const { output, outputFileSystem = fs } = opts;
 
  const appConfig = getAppConfig({ srcDir, outputFileSystem });
 
  const { app } = appConfig;
  let appJson = {};
 
  if (output && !outputFileSystem.existsSync(output)) {
    outputFileSystem.mkdirSync(output, { recursive: true });
  }
 
  // tabBar
  const { tabBar = {} } = app;
  if (tabBar) {
    const { items = [] } = tabBar;
    items.map((tab) => {
      const { pagePath = '' } = tab;
      let match = pagePath.match(/([^?]+)(\?.+)?/);
      let [, pagePathWithOutQs = pagePath] = match || [];
      if (pagePathWithOutQs) {
        if (pagePathWithOutQs[0] === '/') {
          pagePathWithOutQs = pagePathWithOutQs.slice(1);
        }
      }
      const normalizedTab = tab;
      normalizedTab.tag = pagePathWithOutQs;
      normalizedTab.launchParamsTag = pagePathWithOutQs;
      const { icon, activeIcon } = tab;
      // copy icon to out folder
      if (icon) {
        copyIcon(icon, srcDir, output, outputFileSystem || fs);
      }
      if (activeIcon) {
        copyIcon(activeIcon, srcDir, output, outputFileSystem || fs);
      }
      return normalizedTab;
    });
 
    const tabBarConfig = {
      ...tabBar,
      items,
    };
    const tabBarJson = JSON.stringify(tabBarConfig, null, 2);
 
    if (output) {
      outputFileSystem.writeFileSync(path.join(output, 'tabBar.json'), tabBarJson);
    }
  }
 
  // launchParams
  const launchParams = {};
  const { window: appWindowConfig } = app;
  const { pages: origPages } = app;
  const usingComponents = {};
  let components = {};
  const pages = origPages.filter((val, i, self) => self.indexOf(val) === i);
  pages.forEach((page) => {
    const pageConfig = appConfig[page];
    if (pageConfig.usingComponents) {
      resolveComponentConfig(pageConfig, usingComponents, components, srcDir, page);
      usingComponents[page] = pageConfig.usingComponents;
      delete pageConfig.usingComponents;
    }
    launchParams[page] = {
      ...appWindowConfig,
      ...pageConfig,
    };
  });
  // removes duplicates
  appJson.window = appWindowConfig;
  appJson.pages = pages;
 
  appJson.usingComponents = usingComponents;
  appJson.componentConfig = components;
  appJson.launchParams = launchParams;
  // write to file
  if (output) {
    outputFileSystem.writeFileSync(
      path.join(output, 'appConfig.json'),
      JSON.stringify(appJson, null, 2),
    );
  }
  // generateAppConfig and Webpack compiler run in different instance
  // so we need to save appConfig to process so in js-loader, we can read it.
  process.env.MINIAPP_CONFIG = JSON.stringify(appJson);
  return appJson;
}
