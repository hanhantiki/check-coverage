var Rating;
(function (Rating) {
    Rating["NotSet"] = "";
    Rating["Poor"] = "Poor";
    Rating["NeedsImprovement"] = "Needs Improvement";
    Rating["Good"] = "Good";
})(Rating || (Rating = {}));
const Colors = {
    [Rating.Poor]: '#ed5d4c',
    [Rating.NeedsImprovement]: '#f3a63a',
    [Rating.Good]: '#5ecb74',
};
function calculateRating(value, good, poor) {
    if (!value && value !== 0)
        return Rating.NotSet;
    if (value > poor)
        return Rating.Poor;
    if (value > good)
        return Rating.NeedsImprovement;
    return Rating.Good;
}
function getRating(name, value) {
    switch (name) {
        case 'LCP':
            return calculateRating(value, 2500, 4000);
        case 'FID':
            return calculateRating(value, 100, 300);
        case 'CLS':
            return calculateRating(value, 0.1, 0.25);
        // case 'FCP':
        //   return calculateRating(value, 2000, 4000); // Page Speed Insights is 1000 and 3000, lighthouse and web.dev does 2000 and 4000
        // case 'TTFB':
        //   return calculateRating(value, 500, 1500); // CrUX Data Studio report says NI is 500ms to 1500ms
        default:
            return Rating.NotSet;
    }
}
function logWebVitals(event) {
    const { name, delta } = event || {};
    const rating = getRating(name, delta);
    if (rating) {
        console.log(`%c[WebVitals] ${name}: ${delta} - ${rating}`, `color: ${Colors[rating]}`);
    }
}
function sendToGA(event) {
    const { id, name, value, delta } = event || {};
    logWebVitals(event);
    const webVitals = {
        name,
        id,
        value,
        delta,
        valueRounded: Math.round(name === 'CLS' ? value * 1000 : value),
        deltaRounded: Math.round(name === 'CLS' ? value * 1000 : value),
    };
    window.dataLayer = window.dataLayer || [];
    window.dataLayer.push({
        event: 'web-vitals',
        webVitals,
    });
}
function default_1() {
    Promise.resolve().then(() => require('web-vitals')).then(({ getFCP, getTTFB, getCLS, getFID, getLCP }) => {
        // google did not recommend to report webvitals on every change
        // https://github.com/GoogleChrome/web-vitals#report-the-value-on-every-change
        getTTFB(sendToGA);
        getCLS(sendToGA, false);
        getFID(sendToGA);
        getLCP(sendToGA, false);
        getFCP(sendToGA, false);
    });
}
exports.default = default_1;
