import React, { forwardRef, useMemo } from 'react';
import { View, TouchableOpacity, ActivityIndicator, StyleSheet } from 'react-native';
import T from 'prop-types';
import pick from 'lodash/pick';
import omit from 'lodash/omit';
 
import theme from '../../utils/theme';
import Text from '../Text';
 
const sizes = ['big', 'medium', 'small'];
const types = ['primary', 'secondary', 'tertiary'];
const textPropStyle = ['fontSize', 'fontWeight', 'color', 'lineHeight'];
 
const Button = forwardRef((props, ref) => {
  const { id, testId, size, type, children, style, disabled, loading, onTap } = props;
  const componentStyles = useMemo(() => {
    const _size = sizes.includes(size) ? size : Button.defaultProps.size;
    const _type = types.includes(type) ? type : Button.defaultProps.type;
    const disabledStyle = disabled || loading ? styles[`${_type}Disabled`] : {};
    const mergedStyle = StyleSheet.flatten([
      styles.common,
      styles[_size],
      styles[_type],
      disabledStyle,
      style,
    ]);
    const button = omit(mergedStyle, textPropStyle);
    const text = pick(mergedStyle, textPropStyle);
    const loadingStyle = styles[`${_type}Loading`];
 
    return {
      button,
      text,
      loading: loadingStyle,
    };
  }, [size, style, type, disabled, loading]);
 
  const renderChildren = () => {
    const elements = Array.isArray(children) ? children : [children];
    const components = React.Children.map(elements, (child) => {
      if (React.isValidElement(child)) {
        return React.cloneElement(child, {
          style: { ...componentStyles.text, ...child.props.style },
          disabled: disabled || loading,
        });
      }
      return (
        <Text style={componentStyles.text} disabled={disabled || loading}>
          {child}
        </Text>
      );
    });
    return components;
  };
 
  const onPress = (e) => {
    if (!disabled && !loading && typeof onTap === 'function') {
      onTap(e);
    }
  };
 
  return (
    <TouchableOpacity
      nativeID={id}
      testID={testId}
      ref={ref}
      activeOpacity={1}
      disabled={disabled || loading}
      onPress={onPress}
    >
      <View style={componentStyles.button}>
        {!!loading && (
          <ActivityIndicator
            size="small"
            color={componentStyles.loading.color}
            style={componentStyles.loading}
          />
        )}
        {renderChildren()}
      </View>
    </TouchableOpacity>
  );
});
 
const styles = StyleSheet.create({
  common: {
    borderRadius: theme['border-radius-rounded-4px'],
    alignItems: 'center',
    alignSelf: 'flex-start',
    flexDirection: 'row',
    justifyContent: 'center',
  },
  big: {
    fontSize: theme['font-size-medium'],
    fontWeight: theme['font-weight-medium'],
    lineHeight: theme['font-height-medium'],
    paddingVertical: 8,
    paddingHorizontal: 16,
  },
  medium: {
    fontSize: theme['font-size-medium'],
    fontWeight: theme['font-weight-medium'],
    lineHeight: theme['font-height-medium'],
    paddingVertical: 4,
    paddingHorizontal: 12,
  },
  small: {
    fontSize: theme['font-size-small'],
    fontWeight: theme['font-weight-medium'],
    lineHeight: theme['font-height-small'],
    paddingVertical: 4,
    paddingHorizontal: 8,
  },
  primary: {
    backgroundColor: theme.brand,
    color: theme.white,
    borderWidth: 1,
    borderColor: theme.brand,
  },
  secondary: {
    backgroundColor: theme.white,
    color: theme.brand,
    borderWidth: 1,
    borderColor: theme.brand,
  },
  tertiary: {
    backgroundColor: theme.transparent,
    color: theme.brand,
    borderWidth: 1,
    borderColor: theme.transparent,
  },
  primaryDisabled: {
    backgroundColor: theme.gray20,
    color: theme.white,
    borderColor: theme.gray20,
  },
  secondaryDisabled: {
    backgroundColor: theme.white,
    color: theme.gray40,
    borderColor: theme.gray20,
  },
  tertiaryDisabled: {
    backgroundColor: theme.transparent,
    color: theme.gray40,
    borderColor: theme.transparent,
  },
  primaryLoading: {
    color: theme.white,
    marginRight: 8,
  },
  secondaryLoading: {
    color: theme.gray20,
    marginRight: 8,
  },
  tertiaryLoading: {
    color: theme.gray20,
    marginRight: 8,
  },
});
 
Button.propTypes = {
  size: T.oneOf(sizes),
  type: T.oneOf(types),
  children: T.oneOfType([T.node, T.string]),
  style: T.oneOfType([T.object, T.array]),
  disabled: T.bool,
  loading: T.bool,
  testId: T.string,
  id: T.string,
  onTap: T.func,
};
 
Button.defaultProps = {
  size: 'big',
  type: 'primary',
  children: '',
  style: {},
  disabled: false,
  loading: false,
};
 
export default Button;
