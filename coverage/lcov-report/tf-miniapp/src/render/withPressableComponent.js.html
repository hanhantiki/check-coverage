import React from 'react';
import { PanResponder, Platform, TouchableWithoutFeedback } from 'react-native';
import normalizeNativeEvent from './utils/normalizeNativeEvent';
 
const LONG_PRESS_DELAY = 2000;
const TOUCH_THRESHOLD = 20;
 
function pressableComponent(WrappedComponent) {
  return class extends React.PureComponent {
    constructor(props) {
      super(props);
      this.panResponder = this.createPanResponder();
    }
 
    clearLongTapTimeout() {
      this.firedLongTap = false;
      clearTimeout(this.onLongPressTimeout);
    }
 
    _handleTouchEnd(nativeEvent) {
      this.isTouchDown = false;
      const { onTouchEnd, catchTouchEnd, onTap, catchTap } = this.props;
      const touchEnd = onTouchEnd || catchTouchEnd;
      const tap = onTap || catchTap;

    _handleTouchStart(nativeEvent) {
      this.isTouchDown = true;
      const { onTouchStart, catchTouchStart } = this.props;
      const t = onTouchStart || catchTouchStart;
      const e = normalizeNativeEvent(nativeEvent);
      t && t(e);
      this.onLongPressTimeout = setTimeout(() => {
        const { onLongTap, catchLongTap } = this.props;
        const tap = onLongTap || catchLongTap;
        tap && tap(e);
        this.firedLongTap = true;
      }, LONG_PRESS_DELAY);
    }
 
    _handleTouchMove(nativeEvent, gestureState) {
      var dx = 0;
      var dy = 0;
      if (gestureState) {
        dx = gestureState.dx;
        dy = gestureState.dy;
      } else if (this.isTouchDown) {
        dx = nativeEvent.movementX;
        dy = nativeEvent.movementY;
      }
 
      if (Math.abs(dx) > TOUCH_THRESHOLD || Math.abs(dy) > TOUCH_THRESHOLD) {
        this.clearLongTapTimeout();
      }
      if (dx === 0 && dy === 0) {
        return;
      }
      const { onTouchMove, catchTouchMove } = this.props;
      const t = onTouchMove || catchTouchMove;
      t && t(normalizeNativeEvent(nativeEvent));
    }
 
    _handleTouchCancel(nativeEvent) {
      clearTimeout(this.onLongPressTimeout);
      const { onTouchCancel, catchTouchCancel } = this.props;
      const t = onTouchCancel || catchTouchCancel;
      t && t(normalizeNativeEvent(nativeEvent));
    }
 
    createPanResponder() {
      return PanResponder.create({
        onStartShouldSetPanResponder: (evt, gestureState) => true,
        onStartShouldSetPanResponderCapture: (evt, gestureState) => false,
        onMoveShouldSetPanResponder: (evt, gestureState) => {
          const { dx, dy } = gestureState;
          return Math.abs(dx) > TOUCH_THRESHOLD || Math.abs(dy) > TOUCH_THRESHOLD;
        },
        onMoveShouldSetPanResponderCapture: (evt, gestureState) => false,
 
        onPanResponderGrant: (evt, gestureState) => {
          this._handleTouchStart(evt.nativeEvent);
        },
        onPanResponderMove: (evt, gestureState) => {
          this._handleTouchMove(evt, gestureState);
        },
        onPanResponderTerminationRequest: (evt, gestureState) => true,
        onPanResponderRelease: (evt, gestureState) => {
          this._handleTouchEnd(evt.nativeEvent);
        },
        onPanResponderTerminate: (evt, gestureState) => {
          this._handleTouchCancel(evt.nativeEvent);
        },
        onShouldBlockNativeResponder: (evt, gestureState) => {
          return false;
        },
      });
    }
 
    _getEventProps(props) {
      return Object.keys(props).filter((p) => p.match(/^(on|catch)(Touch|Tap|LongTap)[A-Z]*/));
    }
 
    normalizeProps() {

      deletesProps.forEach((e) => delete _props[e]);
      return _props;
    }
 
    maybeThisEvent(eventName, e) {

        return p.match(regex);

      if (thisEvent && eventName.startsWith('on')) {
        this.props[thisEvent].call(this, e);
      }

                _oldEvent(e);
                this.maybeThisEvent(eventName, e);
              };
            }
            return React.cloneElement(child, childProps);
          }
        }
        return child;

        return (
          <TouchableWithoutFeedback onPress={this._handleTouchEnd.bind(this)}>
            <WrappedComponent {...this.normalizeProps()}>{normalizedChildren}</WrappedComponent>
          </TouchableWithoutFeedback>

      }
      return <WrappedComponent {...this.normalizeProps()}>{normalizedChildren}</WrappedComponent>;
    }
  };
