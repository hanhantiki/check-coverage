// import Config from 'react-native-config';
import ContainerInjector from './ContainerInjector';
import getLogger from './logging';
 
let currentRuntime = null;
export const setContainerRuntime = (runtime) => {
  currentRuntime = runtime;
};
 
export const getContainerRuntime = () => {
  return currentRuntime;
};
 
export default class ContainerRuntime {
  constructor(provider) {
    this.logger = getLogger(this.constructor.name);
    this.appRootQuery = 'TikiMiniApp'; // appName in ReactNative or Root Element in Web
    this.isAppStarting = false;
    this.provider = provider;
    this.injector = new ContainerInjector(provider);
    this.injector.setRuntimeContainer(this);
    this.appPendingQueue = [];
 
    setContainerRuntime(this);
  }
 
  startApp(opts = {}, reboot = false) {
    return new Promise((resolve) => {
      if (this.isAppStarting) {
        this.appPendingQueue.push({ resolve, reboot });
      } else {
        this.internal_startApp(opts, reboot).then(() => {
          this.internal_checkAppPendingQueue().then(resolve);
        });
      }
    });
  }
 
  async internal_checkAppPendingQueue() {
    if (this.appPendingQueue.length > 0) {
      const reboot = this.appPendingQueue.reduce((ret, { reboot: _re }) => ret || _re, false);
      while (this.appPendingQueue.length > 0) {
        await this.internal_startApp(reboot);
        this.appPendingQueue.pop().resolve();
      }
    }
  }
 
  // private
  async internal_startApp({
    onReady,
    onAppStarted,
    onDestroyApp,
    appMeta,
    onNavigateToMiniApp,
    onNavigateBackMiniApp,
  }) {
    this.internal_destroyApp();
    this.isAppStarting = true;
    this.onDestroyApp(onDestroyApp);
    this.onNavigateToMiniApp(onNavigateToMiniApp);
    this.onNavigateBackMiniApp(onNavigateBackMiniApp);
    try {
      // create AppRuntime
      if (!this.appRuntime) {
        this.appRuntime = this.injector.createAppRuntime(this.appRootQuery);
      }
 
      // we need frameworkVersion and frameworkCdnBaseDomain
      // because we will load tf-miniapp.worker.js file from server
      const {
        appId,
        cdnBaseUrl,
        appName,
        sha256,
        url,
        frameworkVersion,
        frameworkCdnBaseDomain,
      } = appMeta;
      // get AppMeta
      this.appMeta = {
        appId,
        appName,
        cdnBaseUrl,
        frameworkVersion,
        frameworkCdnBaseDomain,
        sha256,
        launchParams: {
          enableTabBar: 'YES',
          allowsBounceVertical: 'YES',
          defaultTitle: '',
          url,
          pullRefresh: false,
          transparentTitle: 'none',
        },
      };
      // create RootApp
      const app = this.appRuntime.createMiniApp(this.appMeta);
      // notify entry for mount AppComponent
      if (onReady) {
        onReady(app);
      }
      // prepare MiniApp (donwload datasource, setup miniapp env,...);
      await app.prepare();
      // bootstrap MiniApp
      await app.bootstrap();
      this.app = app;
 
      if (onAppStarted) {
        onAppStarted();
      }
      this.logger.log(`App ${app.getId()} started`);
    } catch (e) {
      this.logger.error(e.message);
    }
    this.isAppStarting = false;
  }
 
  getInjector() {
    return this.injector;
  }
 
  onDestroyApp(handler) {
    this.destroyAppHandler = handler;
  }
 
  onNavigateToMiniApp(handler) {
    this.onNavigateToMiniAppHandler = handler;
  }
 
  onNavigateBackMiniApp(handler) {
    this.onNavigateBackMiniAppHandler = handler;
  }
 
  navigateToMiniApp(request, callback) {
    if (this.onNavigateToMiniAppHandler) {
      this.onNavigateToMiniAppHandler(request, callback);
    }
  }
 
  navigateBackMiniApp(request, callback) {
    if (this.onNavigateBackMiniAppHandler) {
      this.onNavigateBackMiniAppHandler(request, callback);
    }
  }
 
  destroyApp() {
    this.internal_destroyApp();
    if (this.destroyAppHandler) {
      this.destroyAppHandler();
    }
  }
 
  internal_destroyApp() {}
 
  registerNativeEvents() {
    // TODO implement this for online or offline
  }
}
