import React from 'react';
import { StyleSheet, Platform } from 'react-native';
import { WebView as RNWebView } from 'react-native-webview';
import FreeRenderContext from '../FreeRenderContext';
 
let webviewIdSeq = 0;
 
export default class WebView extends React.PureComponent {
  static contextType = FreeRenderContext;
 
  constructor(props) {
    super(props);
    this.type = Platform.OS === 'web' ? 'iframe' : 'webview';
    this.callPool = {};
    this.html = null;
    this.id = `webview_${webviewIdSeq++}`;
    this.onMessageHandler = props.onMessage;
  }
 
  // TODO inject user-agent
  injectBridge = () => {
    const JSBridge = {
      call: (api, params, callback) => {
        switch (api) {
          case 'postWebViewMessage':
            // dispatch to worker
            const { renderComponent } = this.context;
            const app = renderComponent.getApp();
            const worker = app.getWorker();
            const { api: workerApi, params: workerApiParams } = params;
            if (workerApi === 'postMessage') {
              if (this.onMessageHandler) {
                // the last arg is native event.
                this.onMessageHandler(workerApiParams, {});
              }
            } else {
              worker.callApi(workerApi, workerApiParams, callback);
            }
            break;
          case 'onWebViewUrlChange':
            break;
        }
      },
    };
 
    const mapApiWithCallId = (apis) => {
      return Object.keys(apis).reduce((map, api) => {
        if (typeof apis[api] === 'function') {
          const callId = this.generateCallId();
          this.callPool[callId] = apis[api];
          map[api] = callId;
          return map;
        }
        if (typeof apis[api] === 'object') {
          map[api] = mapApiWithCallId(apis[api]);
        }
        return map;
      }, {});
    };
    const apiToCallId = mapApiWithCallId({ JSBridge });
    this.injectJavaScript(`
      var __KERNEL_PORT__ = {
        onmessage: () => {},
        postMessage: (message) => {
          if (window.ReactNativeWebView) {
            window.ReactNativeWebView.postMessage(message);
          } else if (window.postMessageToRuntime) {
            window.postMessageToRuntime(message);
          }
        }
      };
      var __ENV__ = ${JSON.stringify(apiToCallId)}; 
      document.dispatchEvent(new Event('JSBridgeReady'));
    `);
  };
 
  injectJavaScript(code) {
    if (this.type === 'iframe') {
      this.webview.frameRef.contentWindow.eval(code);
    } else {
      this.webview.injectJavaScript(`${code};true`);
    }
  }
 
  generateCallId() {
    return Math.floor(Math.random() * 1000000);
  }
 
  injectFramework(html) {
    if (html.includes('https://tiki/tf-miniapp.webview.js')) {
      return html.replace('https://tiki/tf-miniapp.webview.js', this.frameworkUrl);
    }
    const headIdx = html.indexOf('<head>');
    if (headIdx !== -1) {
      return `${html.substring(0, headIdx + 6)}<script src="${
        this.frameworkUrl
      }"></script>${html.substring(headIdx + 6)}`;
    }
    const htmlIdx = html.indexOf('<html>');
    return `${html.substring(0, htmlIdx + 6)}<head><script src="${
      this.frameworkUrl
    }"></script></head>${html.substring(htmlIdx)}`;
  }
 
  onWebViewMessage = ({ nativeEvent }) => {
    const { data: json } = nativeEvent;
    try {
      const data = JSON.parse(json);
      const { callId, callbackId, params } = data;
      if (callId) {
        if (this.callPool[callId]) {
          const { method, args } = params;
          const argsWithCallback = [...(args || [])];
          if (callbackId) {
            argsWithCallback.push((...callbackArgs) => {
              this.postMessage({
                callbackId,
                method: 'invokeCallback',
                args: callbackArgs,
              });
            });
          }
          if (typeof this.callPool[callId] === 'function') {
            this.callPool[callId](...argsWithCallback);
          } else if (
            typeof this.callPool[callId] === 'object' &&
            method &&
            typeof this.callPool[callId][method] === 'function'
          ) {
            this.callPool[callId][method](...argsWithCallback);
          }
        }
        return;
      }
    } catch (e) {}
  };
 
  postMessage = (data) => {
    const message = { data: JSON.stringify(data) };
    const injectedScript = `
    __KERNEL_PORT__.onmessage(${JSON.stringify(message)});
    `;
    this.injectJavaScript(injectedScript);
  };
 
  async requestWebViewContent(uri) {
    if (Platform.OS === 'web') {
      const params = {
        headers: {
          'user-agent': 'TikiNative',
        },
        url: uri,
      };
      const response = await fetch(
        `/proxy/request?params=${encodeURIComponent(JSON.stringify(params))}`,
      );
      return await response.text();
    }
    const response = await fetch(uri, {
      headers: {
        // TODO: need better user-agent
        // 'user-agent': 'Mozilla/5.0 (iPhone; CPU iPhone OS 13_2_3 like Mac OS X) AppleWebKit/605.1.15 (KHTML, like Gecko) Version/13.0.3 Mobile/15E148 Safari/604.1'
        'user-agent': 'TikiNative',
      },
    });
    return await response.text();
  }
 
  async componentDidMount() {
    const { renderComponent } = this.context;
    const app = renderComponent.getApp();
 
    this.frameworkUrl = app.dataSource.resolvePathToCdn('tf-miniapp.webview.js');
    const { source = {} } = this.props;
    const { uri } = source;
    let html = await this.requestWebViewContent(uri);
    if (this.type === 'iframe') {
      const headIdx = html.indexOf('<head>');
      if (headIdx !== -1) {
        const { protocol, host } = window.location;
        const proxyHost = `${protocol}//${host}`;
        // TODO change user agent TikiNative here
        // TODO suport POST request
        html = `${html.substring(0, headIdx + 6)}
          <base href=${JSON.stringify(uri)} />
          <script>(function () {
            Object.defineProperty(navigator, 'userAgent', {
              enumerable: true,
              configurable: true,
              value: 'TikiNative'
            });
          })();
          var proxied = window.XMLHttpRequest.prototype.open;
          window.XMLHttpRequest.prototype.open = function() {
            const args = [].slice.call(arguments);
            const method = args.shift();
            const url = args.shift();
            if (url.startsWith(${JSON.stringify(proxyHost)})) {
              return proxied.apply(this, [].slice.call(arguments));
            }
            const params = { url: url, method: method, headers: {} };
            args.unshift(${JSON.stringify(
              proxyHost,
            )} + '/proxy/request?params=' + encodeURIComponent(JSON.stringify(params)));
            args.unshift('GET');
            return proxied.apply(this, [].slice.call(args));
          };
          history.pushState = function() { }
          history.replaceState = function() { }
          </script>
        ${html.substring(headIdx + 6)}`;
      }
    }
    this.html = this.injectFramework(html);
    renderComponent.addWebView(this);
  }
 
  async getFrameworkBundleContent() {
    const frameworkBundle = await fetch(this.frameworkUrl);
    const content = await frameworkBundle.text();
    return content;
  }
 
  getComponent() {
    if (this.html === null) {
      return null;
    }
    const { source = {} } = this.props;
    const { uri } = source;
    const webviewSource = { html: this.html, baseUrl: uri };
 
    return (
      <RNWebView
        key={this.id}
        originWhitelist={['https://*', 'http://localhost*']}
        onMessage={(...args) => {
          if (this.type !== 'iframe') {
            this.onWebViewMessage(...args);
          }
        }}
        source={webviewSource}
        ref={(ref) => {
          this.webview = ref;
        }}
        onLoad={async () => {
          if (this.type === 'iframe') {
            this.webview.frameRef.contentWindow.postMessageToRuntime = (message) => {
              this.onWebViewMessage({
                nativeEvent: {
                  data: message,
                },
              });
            };
          }
          this.injectBridge();
        }}
        style={s.container}
        decelerationRate="normal"
        showsHorizontalScrollIndicator={false}
        showsVerticalScrollIndicator={false}
      />
    );
  }
 
  render() {
    return null;
  }
}
